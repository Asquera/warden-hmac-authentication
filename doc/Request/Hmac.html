def sign(env)
  signer = HMAC::Signer.new
  url = env[:url]
  headers, url = *signer.sign_request(url, @secret, @options)
    
  env[:request_headers] = (env[:request_headers] || {}).merge(headers)
  env[:url] = Addressable::URI.parse(url)
  env
end
